```python3
data = [1,2,3,4,5]
newdata = []
for d in data:
    newdata.append(d * 2)
print(newdata)
```
# リスト内包表記で
```python3
newdata = [d * 2 for d in data]
print(newdata)</pre>

newdata = []
for d in data:
    if d % 2 == 0:
        newdata.append(d * 2)
print(newdata)
```

# if文付きのリスト処理
```python3
newdata = [d * 2 for d in data if d % 2 == 0]
print(newdata)
```

# セット内包表記
```python3
a_set = {number for number in range(20) if number % 3 == 0}
print(a_set)
```

# 辞書の包括表記
```python3
word = "letters"
letter_count = {letter:word.count(letter) for letter in set(word)}
print(letter_count)
```
